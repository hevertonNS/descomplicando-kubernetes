# Meu Primeiro Pod no Kubernetes
# Objetivo: Criar um servidor Nginx simples para testar o cluster.
apiVersion: v1
kind: Pod
metadata:
  name: meu-primeiro-pod
  labels:
    app: web-server
    type: frontend
spec:
  containers:
    - name: nginx-container
      image: nginx:latest
      ports:
        - containerPort: 80
